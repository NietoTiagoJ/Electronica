}

	this->titulo= t;
	this->director = direc;
	this->duracion = d;

}

}

	cout<<this->duracion<<endl;

#include <iostream>
#include <vector>

// Función para encontrar el mayor número en un vector de enteros
int encontrarMayor(const std::vector<int>& vec) {
    int mayor = vec[0];
    for (std::size_t i = 1; i < vec.size(); i++) {
        if (vec[i] > mayor) {
            mayor = vec[i];
        }
    }
    return mayor;
}

El siguiente programa utiliza una macro para encontrar el valor máximo entre dos números. El programa debe ser modificado para reemplazar esta macro por una función inline que acepte los tipo de dato: enteros, flotantes o caracteres, y devuelva el valor máximo. Además, la función debe ser probada en el main con diferentes tipos de datos. Puedes usar plantilla o sobrecarga de función.

#include <iostream>

// Macro para encontrar el valor máximo entre dos números
#define MAXIMO(a, b) ((a) > (b) ? (a) : (b))

int main() {
    int num1 = 3, num2 = 7;
    float num3 = 4.5, num4 = 2.3;
    char char1 = 'a', char2 = 'z';

    std::cout << "Máximo entre " << num1 << " y " << num2 << " es: " << MAXIMO(num1, num2) << std::endl;
    std::cout << "Máximo entre " << num3 << " y " << num4 << " es: " << MAXIMO(num3, num4) << std::endl;
    std::cout << "Máximo entre '" << char1 << "' y '" << char2 << "' es: " << MAXIMO(char1, char2) << std::endl;

    return 0;
}
